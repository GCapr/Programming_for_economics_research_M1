3.1 Handling Missing Data

Missing data is ubiquitous in real-world datasets. Understanding why data is missing is crucial for choosing the appropriate handling strategy. There are three main types of missingness:

Type Description Example
MCAR Missing Completely At Random Survey responses lost due to computer glitch
MAR Missing At Random (conditional on observed) Income missing more often for older respondents
MNAR Missing Not At Random High earners refuse to report income 
# Python: Handling missing data
import pandas as pd
import numpy as np

# Check for missing values
print(df.isnull().sum())           # Count by column
print(df.isnull().mean() * 100)   # Percentage missing

# Visualize missing patterns
import missingno as msno
msno.matrix(df)
msno.heatmap(df)  # Correlation of missingness

# Drop missing values
df_complete = df.dropna()                    # Drop any row with NA
df_subset = df.dropna(subset=['income'])    # Drop if income is missing

# Fill missing values
df['income'].fillna(df['income'].mean(), inplace=True)  # Mean imputation
df['income'].fillna(df['income'].median(), inplace=True) # Median imputation
df['income'].fillna(method='ffill', inplace=True)      # Forward fill

# Group-wise imputation
df['income'] = df.groupby('education')['income'].transform(
    lambda x: x.fillna(x.median())
)

# Create missing indicator
df['income_missing'] = df['income'].isnull().astype(int)

library(naniar)    # For missing data visualization

# Replace missing values
df <- df %>%
  mutate(
    income = replace_na(income, mean(income, na.rm = TRUE))
  )

# Group-wise imputation
df <- df %>%
  group_by(education) %>%
  mutate(
    income = ifelse(is.na(income), median(income, na.rm = TRUE), income)
  ) %>%
  ungroup()

# Create missing indicator
df <- df %>%
  mutate(income_missing = as.integer(is.na(income)))

3.2 Detecting and Treating Outliers

Outliers can significantly affect your analysis. Before removing them, always investigate why they exist‚Äîthey might be data errors, or they might be genuine extreme values that contain important information. 

# Python: Detecting and treating outliers
import pandas as pd
import numpy as np
from scipy import stats

# Method 1: IQR method
Q1 = df['income'].quantile(0.25)
Q3 = df['income'].quantile(0.75)
IQR = Q3 - Q1
lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR

outliers = df[(df['income'] < lower) | (df['income'] > upper)]
print(f"Outliers: {len(outliers)}")

# Method 2: Z-score
df['z_score'] = np.abs(stats.zscore(df['income']))
outliers_z = df[df['z_score'] > 3]

# Treatment: Winsorize (cap at percentiles)
p01 = df['income'].quantile(0.01)
p99 = df['income'].quantile(0.99)
df['income_win'] = df['income'].clip(p01, p99)

# Using scipy
from scipy.stats.mstats import winsorize
df['income_win'] = winsorize(df['income'], limits=[0.01, 0.01])
* Stata: Detecting and treating outliers

* Visual inspection
histogram income
graph box income

* Identify using percentiles
summarize income, detail
local p1 = r(p1)
local p99 = r(p99)

* Winsorize at 1st and 99th percentiles
winsor2 income, cuts(1 99) suffix(_win)  // ssc install winsor2

* Manual winsorizing
egen p1 = pctile(income), p(1)
egen p99 = pctile(income), p(99)
gen income_win = income
replace income_win = p1 if income < p1
replace income_win = p99 if income > p99

* Flag outliers
gen outlier = (income < p1 | income > p99)
# R: Detecting and treating outliers
library(tidyverse)

# Method 1: IQR method
Q1 <- quantile(df$income, 0.25, na.rm = TRUE)
Q3 <- quantile(df$income, 0.75, na.rm = TRUE)
IQR_val <- Q3 - Q1
lower <- Q1 - 1.5 * IQR_val
upper <- Q3 + 1.5 * IQR_val

outliers <- df %>% filter(income < lower | income > upper)

# Winsorize function
winsorize <- function(x, probs = c(0.01, 0.99)) {
  limits <- quantile(x, probs, na.rm = TRUE)
  x[x < limits[1]] <- limits[1]
  x[x > limits[2]] <- limits[2]
  return(x)
}

df <- df %>%
  mutate(
    income_win = winsorize(income),
    outlier = income < lower | income > upper
  )

3.3 String Manipulation

Text data often requires cleaning‚Äîstandardizing case, removing whitespace, extracting patterns, and more. 

# Python: String manipulation
import pandas as pd

# Case conversion
df['name_lower'] = df['name'].str.lower()
df['name_upper'] = df['name'].str.upper()
df['name_title'] = df['name'].str.title()

# Whitespace
df['name'] = df['name'].str.strip()          # Remove leading/trailing
df['name'] = df['name'].str.replace(r'\s+', ' ', regex=True)  # Collapse spaces

# Extract patterns with regex
df['zip'] = df['address'].str.extract(r'(\d{5})')
df['state'] = df['address'].str.extract(r', ([A-Z]{2}) \d')

# Replace values
df['status'] = df['status'].replace({'M': 'Married', 'S': 'Single'})

# Split strings
df[['first', 'last']] = df['full_name'].str.split(' ', n=1, expand=True)

# Contains / startswith / endswith
df['is_dr'] = df['name'].str.startswith('Dr.')

* Extract substrings

* Regular expressions
gen has_number = regexm(name, "[0-9]")
gen zip = regexs(1) if regexm(address, "([0-9]{5})")

* Replace

    # Case conversion

    name_upper = str_to_upper(name),
    name_title = str_to_title(name),
    
    # Whitespace
    name = str_trim(name),                   # Remove leading/trailing

    # Detect patterns

  )

# Split strings

  separate(full_name, into = c("first", "last"), sep = " ", extra = "merge")

3.4 Working with Dates

# Python: Working with dates
import pandas as pd

# Parse dates
df['date'] = pd.to_datetime(df['date_string'])
df['date'] = pd.to_datetime(df['date_string'], format='%Y-%m-%d')
df['date'] = pd.to_datetime(df['date_string'], format='%m/%d/%Y')

# Extract components
df['year'] = df['date'].dt.year
df['month'] = df['date'].dt.month
df['day'] = df['date'].dt.day
df['weekday'] = df['date'].dt.day_name()
df['quarter'] = df['date'].dt.quarter

# Date arithmetic
df['days_since'] = (pd.Timestamp.today() - df['date']).dt.days
df['next_month'] = df['date'] + pd.DateOffset(months=1)

# Filter by date
df_2020 = df[df['date'].dt.year == 2020]
df_recent = df[df['date'] >= '2020-01-01']

# R: Working with dates (lubridate)
library(tidyverse)
library(lubridate)

df <- df %>%
  mutate(
    # Parse dates (lubridate auto-detects many formats)

df_recent <- df %>% filter(date >= ymd("2020-01-01"))

3.5 Data Validation

Before analysis, always validate your data for consistency and logical errors. 

# Python: Data validation
import pandas as pd

# Check for duplicates
print(f"Duplicates: {df.duplicated().sum()}")
duplicates = df[df.duplicated(subset=['id'], keep=False)]

# Check value ranges
assert df['age'].between(0, 120).all(), "Invalid ages found"
assert df['income'].min() >= 0, "Negative income found"

# Check categorical values
valid_genders = ['M', 'F', 'Other']
invalid = df[~df['gender'].isin(valid_genders)]

# Check logical consistency
errors = df[df['birth_year'] > df['hire_year']]

# Summary report
def validate_data(df):
    report = {
        'total_rows': len(df),
        'duplicates': df.duplicated().sum(),
        'missing_pct': df.isnull().mean().to_dict()
    }
    return report

# R: Data validation
library(tidyverse)

# Check for duplicates
sum(duplicated(df))
df %>% filter(duplicated(id) | duplicated(id, fromLast = TRUE))

# Check value ranges with assertions
stopifnot(all(df$age >= 0 & df$age <= 120, na.rm = TRUE))
stopifnot(all(df$income >= 0, na.rm = TRUE))

# Check categorical values
valid_genders <- c("M", "F", "Other")
df %>% filter(!gender %in% valid_genders)

# Logical consistency
df %>% filter(birth_year > hire_year)

# Summary function
validate_data <- function(df) {
  list(
    total_rows = nrow(df),
    duplicates = sum(duplicated(df)),
    missing = colMeans(is.na(df))
  )
}
